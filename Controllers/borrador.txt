
    function getDatosImagen(){
        //session_start(); // ver admin
        $id_pedido = $_POST["id_pedido"];
        $this->agregarImagenes($_FILES['image'], $id_pedido);
        header('Location: ' . BASE_URL . 'vermas/' . $id_pedido);
    }

    function agregarImagenes($images, $id_pedido) {
        for ($i=0; $i < sizeof($images['name']); $i++){
            $extension = pathinfo($images['name'][$i], PATHINFO_EXTENSION);
            if ($extension == "jpeg" || $extension == "jpg" || $extension == "png" || $extension == "gif"){   
                $nombre = $images['name'][$i];
                $nombre_tmp = $images['tmp_name'][$i];
                $imagen = $this->moveFile($nombre, $nombre_tmp);
                $this->imagenesModel->insertarImagenes($id_pedido, $imagen);
            }    
        }
    }

    // MOVE FILE
    private function moveFile($nombre, $tmp) {
        $filepath = "images_tmp/" . uniqid() . "." . strtolower(pathinfo($nombre, PATHINFO_EXTENSION)); // strtolower -> caracteres en minúscula  
        move_uploaded_file($tmp, $filepath);
        return $filepath;
    }
